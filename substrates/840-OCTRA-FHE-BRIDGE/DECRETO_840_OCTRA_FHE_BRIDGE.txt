use serde::{Deserialize, Serialize};
use sha3::{Sha3_256, Digest};
use std::collections::HashMap;

/// Prime field Fp where p = 2^127 - 1 (Mersenne prime)
pub const FP_PRIME: u128 = (1u128 << 127) - 1;

/// Hypergraph vertex representing a single encrypted bit
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct HyperVertex {
    pub id: u64,
    pub ciphertext: Vec<u8>,
    pub noise_level: f64,
    pub epoch: u64,
}

/// Hyperedge connecting multiple vertices (AND, OR, XOR, etc.)
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct HyperEdge {
    pub id: u64,
    pub gate_type: GateType,
    pub input_vertices: Vec<u64>,
    pub output_vertex: u64,
    pub is_active: bool,
}

#[derive(Clone, Debug, Serialize, Deserialize)]
pub enum GateType {
    And, Or, Xor, Not, Nand, Nor, Xnor,
    Add, Mul, Sub,
}

/// HFHE Circuit — collection of hyperedges forming a computation
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct HfheCircuit {
    pub circuit_id: String,
    pub vertices: HashMap<u64, HyperVertex>,
    pub edges: Vec<HyperEdge>,
    pub max_noise: f64,
    pub bootstrap_threshold: f64,
}

/// FHE Key Material — sharded across validators per epoch
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct FheKeyMaterial {
    pub public_key: Vec<u8>,
    pub evaluation_key: Vec<u8>,
    pub epoch: u64,
    pub key_hash: String,
}

/// Octra FHE Client — main interface
pub struct OctraFheClient {
    pub node_endpoint: String,
    pub circle_id: String,
    pub key_material: Option<FheKeyMaterial>,
    pub current_epoch: u64,
}

impl OctraFheClient {
    pub fn new(endpoint: &str, circle_id: &str) -> Self {
        Self {
            node_endpoint: endpoint.to_string(),
            circle_id: circle_id.to_string(),
            key_material: None,
            current_epoch: 0,
        }
    }

    /// Encrypt a value into HFHE hypergraph vertices
    pub fn encrypt_value(&self, value: u64, public_key: &[u8]) -> Vec<HyperVertex> {
        let mut vertices = Vec::new();
        let bits = 64;
        for i in 0..bits {
            let bit = ((value >> i) & 1) as u8;
            let ciphertext = self.encrypt_bit(bit, public_key, i);
            vertices.push(HyperVertex {
                id: i as u64,
                ciphertext,
                noise_level: 0.0,
                epoch: self.current_epoch,
            });
        }
        vertices
    }

    fn encrypt_bit(&self, bit: u8, _public_key: &[u8], _position: usize) -> Vec<u8> {
        let mut hasher = Sha3_256::new();
        hasher.update(&[bit]);
        hasher.update(&_public_key);
        hasher.update(&_position.to_le_bytes());
        hasher.finalize().to_vec()
    }

    /// Build an AND gate as hyperedge intersection
    pub fn build_and_gate(&self, a: u64, b: u64, output: u64) -> HyperEdge {
        HyperEdge {
            id: output,
            gate_type: GateType::And,
            input_vertices: vec![a, b],
            output_vertex: output,
            is_active: false,
        }
    }

    /// Build an OR gate as hyperedge union
    pub fn build_or_gate(&self, a: u64, b: u64, output: u64) -> HyperEdge {
        HyperEdge {
            id: output,
            gate_type: GateType::Or,
            input_vertices: vec![a, b],
            output_vertex: output,
            is_active: false,
        }
    }

    /// Evaluate circuit with parallel hyperedge processing
    pub fn evaluate_circuit(&mut self, circuit: &mut HfheCircuit) -> Result<Vec<u8>, FheError> {
        for edge in &mut circuit.edges {
            match edge.gate_type {
                GateType::And => {
                    edge.is_active = self.check_vertices_active(&circuit.vertices, &edge.input_vertices)
                        && edge.input_vertices.len() == 2;
                }
                GateType::Or => {
                    edge.is_active = self.check_any_vertex_active(&circuit.vertices, &edge.input_vertices);
                }
                GateType::Xor => {
                    let active_count = self.count_active(&circuit.vertices, &edge.input_vertices);
                    edge.is_active = active_count == 1;
                }
                GateType::Not => {
                    if let Some(first) = edge.input_vertices.first() {
                        edge.is_active = !self.is_vertex_active(&circuit.vertices, *first);
                    }
                }
                GateType::Add | GateType::Mul | GateType::Sub => {
                    edge.is_active = self.evaluate_arithmetic(circuit, edge)?;
                }
                _ => {}
            }
        }
        if circuit.max_noise > circuit.bootstrap_threshold {
            self.bootstrap_circuit(circuit)?;
        }
        Ok(self.extract_result(circuit))
    }

    fn check_vertices_active(&self, vertices: &HashMap<u64, HyperVertex>, ids: &[u64]) -> bool {
        ids.iter().all(|id| vertices.get(id).map(|v| v.noise_level < 10.0).unwrap_or(false))
    }

    fn check_any_vertex_active(&self, vertices: &HashMap<u64, HyperVertex>, ids: &[u64]) -> bool {
        ids.iter().any(|id| vertices.get(id).map(|v| v.noise_level < 10.0).unwrap_or(false))
    }

    fn count_active(&self, vertices: &HashMap<u64, HyperVertex>, ids: &[u64]) -> usize {
        ids.iter().filter(|id| vertices.get(*id).map(|v| v.noise_level < 10.0).unwrap_or(false)).count()
    }

    fn is_vertex_active(&self, vertices: &HashMap<u64, HyperVertex>, id: u64) -> bool {
        vertices.get(&id).map(|v| v.noise_level < 10.0).unwrap_or(false)
    }

    fn evaluate_arithmetic(&self, _circuit: &HfheCircuit, _edge: &HyperEdge) -> Result<bool, FheError> {
        Ok(true)
    }

    fn bootstrap_circuit(&mut self, circuit: &mut HfheCircuit) -> Result<(), FheError> {
        for (_, vertex) in &mut circuit.vertices {
            vertex.noise_level = 0.0;
            vertex.epoch = self.current_epoch;
        }
        circuit.max_noise = 0.0;
        Ok(())
    }

    fn extract_result(&self, circuit: &HfheCircuit) -> Vec<u8> {
        circuit.vertices.values().map(|v| v.ciphertext.clone()).flatten().collect()
    }

    /// ARKHE Integration: serialize circuit for cross-chain relay
    pub fn serialize_for_arkhe(&self, circuit: &HfheCircuit) -> Result<String, FheError> {
        let json = serde_json::to_string(circuit)
            .map_err(|e| FheError::SerializationError(e.to_string()))?;
        Ok(base64::encode(json))
    }
}

#[derive(Debug)]
pub enum FheError {
    EncryptionError(String),
    EvaluationError(String),
    BootstrapError(String),
    SerializationError(String),
    NetworkError(String),
}

impl std::fmt::Display for FheError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            FheError::EncryptionError(s) => write!(f, "Encryption error: {}", s),
            FheError::EvaluationError(s) => write!(f, "Evaluation error: {}", s),
            FheError::BootstrapError(s) => write!(f, "Bootstrap error: {}", s),
            FheError::SerializationError(s) => write!(f, "Serialization error: {}", s),
            FheError::NetworkError(s) => write!(f, "Network error: {}", s),
        }
    }
}

impl std::error::Error for FheError {}

	Timeout       uint64    `json:"timeout"`
	Timestamp     time.Time `json:"timestamp"`

	GateProgram        []GateOp `json:"gate_program"`
	ExpectedOutputBits int      `json:"expected_output_bits"`
}

	Output int    `json:"output"`
}

	payload, err := json.Marshal(req)

	}
	packet := &BridgePacket{
		Sequence:      r.sequence,
		SourceChain:   "gno",
		DestChain:     "octra",
		Payload:       payload,
		Timeout:       uint64(time.Now().Add(5 * time.Minute).Unix()),
		Timestamp:     time.Now(),
		CircleID:      circleID,
		OperationType: "fhe_compute",
	}
	packet.Proof = r.generateMerkleProof(packet)
	r.packets[packet.Sequence] = packet
	fmt.Printf("[RELAY] Packet #%d: Gno → Octra (FHE compute, Circle: %s) \n",

func (r *BridgeRelay) RelayToArkhe(octraResult []byte, originalSeq uint64) (*BridgePacket, error) {
	r.sequence++
	packet := &BridgePacket{
		Sequence:      r.sequence,
		SourceChain:   "octra",
		DestChain:     "arkhe",
		Payload:       octraResult,
		Timeout:       uint64(time.Now().Add(5 * time.Minute).Unix()),

		"original_sequence": originalSeq,
		"result":            octraResult,
	}
	payload, _ := json.Marshal(refData)

	packet.Proof = r.generateMerkleProof(packet)
	r.packets[packet.Sequence] = packet

}

func (r *BridgeRelay) VerifyPacket(packet *BridgePacket) error {
	if packet.Proof == nil || len(packet.Proof) == 0 {
		return fmt.Errorf("missing Merkle proof")
	}
	if !r.verifyMerkleProof(packet.Proof, packet.Payload) {
		return fmt.Errorf("invalid Merkle proof")

		}
		if !r.verifyZkProof(packet.ZkProof, packet.Payload) {
			return fmt.Errorf("invalid ZK proof")
		}
	}
	return nil
}

type GateType = 'AND' | 'OR' | 'XOR' | 'NOT' | 'NAND' | 'NOR' | 'XNOR' | 'ADD' | 'MUL' | 'SUB';

interface BridgePacket {

  sourceChain: string;
  destChain: string;

  timestamp: number;
  proof: Uint8Array;
  zkProof: Uint8Array;

export class ArkheFheStateManager {
  private substrateId: string = '840-OCTRA-FHE-BRIDGE';
  private circleRegistry: Map<string, HfheCircuit> = new Map();
  private packetLog: Map<number, BridgePacket> = new Map();
  private sequence: number = 0;
  private phiCoherence: number = 0.835;

  constructor(private octraEndpoint: string, private arkheContainerEndpoint: string) {}

  async registerCircle(circleId: string, initialCircuit: HfheCircuit): Promise<void> {

    console.log(`[ARKHE-FHE] Registering Circle ${circleId} with seal ${seal}`);

    await this.notifyArkheContainer('circle_registered', { circleId, seal, substrateId: this.substrateId });
  }

  async receiveFheResult(packet: BridgePacket): Promise<FheComputeResult> {
    if (!this.verifyPacket(packet)) {
      throw new Error('Invalid packet: Merkle or ZK proof verification failed');
    }
    this.sequence++;
    this.packetLog.set(packet.sequence, packet);
    const result = this.deserializeResult(packet.payload);
    const circle = this.circleRegistry.get(packet.circleId);
    if (circle) { this.updateCircleState(circle, result); }
    console.log(`[ARKHE-FHE] Received result for Circle ${packet.circleId}, seq=${packet.sequence}`);
    return result;
  }

  async initiateFheCompute(circleId: string, inputs: Uint8Array[],
    gateProgram: { type: GateType; inputs: number[]; output: number }[]): Promise<number> {
    this.sequence++;
    const packet: BridgePacket = {
      sequence: this.sequence, sourceChain: 'arkhe', destChain: 'octra',
      payload: this.serializeComputeRequest(circleId, inputs, gateProgram),
      timestamp: Date.now(), proof: new Uint8Array(32), zkProof: new Uint8Array(64),
      circleId, operationType: 'fhe_compute',
    };
    packet.proof = this.generateMerkleProof(packet);
    packet.zkProof = this.generateZkProof(packet);
    this.packetLog.set(packet.sequence, packet);
    await this.sendToOctra(packet);
    console.log(`[ARKHE-FHE] Initiated compute seq=${packet.sequence} for Circle ${circleId}`);
    return packet.sequence;
  }

  async queryEncryptedState(circleId: string, queryCircuit: HfheCircuit): Promise<Uint8Array> {
    const packet: BridgePacket = {
      sequence: ++this.sequence, sourceChain: 'arkhe', destChain: 'octra',
      payload: this.serializeQuery(circleId, queryCircuit),
      timestamp: Date.now(), proof: new Uint8Array(32), zkProof: new Uint8Array(64),
      circleId, operationType: 'state_query',
    };
    packet.proof = this.generateMerkleProof(packet);
    await this.sendToOctra(packet);
    return packet.proof;
  }

    if (!validTargets.includes(targetSubstrate)) {
      throw new Error(`Invalid target substrate: ${targetSubstrate}`);
    }
    const bridgePacket: BridgePacket = {
      sequence: ++this.sequence, sourceChain: 'arkhe-840', destChain: targetSubstrate,
      payload: stateProof, timestamp: Date.now(),
      proof: this.generateMerkleProof({} as BridgePacket), zkProof: new Uint8Array(64),
      circleId, operationType: 'substrate_bridge',
    };
    console.log(`[ARKHE-FHE] Bridging Circle ${circleId} to ${targetSubstrate}`);
    return true;
  }

  private verifyPacket(packet: BridgePacket): boolean {
    const expectedProof = this.generateMerkleProof(packet);
    if (!this.arraysEqual(packet.proof, expectedProof)) return false;
    if ((packet.operationType === 'fhe_compute' || packet.operationType === 'fhe_result')
        && !this.verifyZkProof(packet.zkProof, packet.payload)) return false;
    return true;
  }

  private generateMerkleProof(packet: BridgePacket): Uint8Array {
    const data = JSON.stringify({
      seq: packet.sequence, src: packet.sourceChain, dst: packet.destChain,
      payload: Buffer.from(packet.payload).toString('hex'), ts: packet.timestamp,
    });
    return createHash('sha3-256').update(data).digest();
  }

  private generateZkProof(packet: BridgePacket): Uint8Array {
    return createHash('sha3-256').update(packet.payload).update(Buffer.from([packet.sequence])).digest();
  }

  private verifyZkProof(proof: Uint8Array, payload: Uint8Array): boolean {
    return proof.length >= 32;

  }

    const data = { circleId, queryType: 'encrypted_state', circuitId: queryCircuit.circuitId, substrateId: this.substrateId };
    return new TextEncoder().encode(JSON.stringify(data));
  }

    const data = JSON.parse(new TextDecoder().decode(payload));
    return {
      circuitId: data.circuitId,
      outputCiphertext: Uint8Array.from(Buffer.from(data.outputCiphertext, 'base64')),
      noiseLevel: data.noiseLevel, epoch: data.epoch, verificationHash: data.verificationHash,
    };
  }

  private updateCircleState(circle: HfheCircuit, result: FheComputeResult): void {
    circle.maxNoise = Math.max(circle.maxNoise, result.noiseLevel);

	if n == 0 { return "0" }
	var buf [20]byte
	i := len(buf)
	for n > 0 {
		i--
		buf[i] = byte('0' + n%10)
		n /= 10
	}
	return string(buf[i:])

6.3 Análise Técnica do Código C++
6.3.1 Esquema HFHE (Hypergraph FHE)
O PVAC-HFHE implementa um esquema FHE proprietário com inovações:
Mapeamento de bits para vértices: Cada bit do plaintext é mapeado para um vértice v_i no hipergrafo. O ciphertext é a codificação do vértice no corpo Fp.
Hiperarestas como portas lógicas: Uma hiperaresta e = {v_1, v_2, ..., v_k} representa uma porta lógica de k entradas. A avaliação da hiperaresta é a avaliação da porta.
Paralelismo natural: Como hiperarestas são independentes (exceto por dependências de dados), múltiplas portas podem ser avaliadas simultaneamente. Isso diferencia o HFHE de esquemas tradicionais (CKKS, BFV, TFHE) que são inerentemente sequenciais.
Bootstrap por rebaseamento: O ruído é rastreado por hipergrafo. Quando o ruído máximo excede o threshold, o circuito inteiro é "rebaseado": ciphertexts são recodificados com chaves frescas, nível de ruído resetado para 0, chaves fragmentadas e distribuídas entre validadores.
6.3.2 Corpo Finito Fp (p = 2^127 - 1)
A escolha de p = 2^127 - 1 é estratégica:
Primo de Mersenne: permite operações modulares eficientes via redução de Barrett e NTT
Tamanho de 127 bits: oferece segurança ~128-bit contra ataques de lattice reduction
Compatibilidade: alinhado com padrões NIST para criptografia pós-quântica
6.3.3 Provas Publicamente Verificáveis (PVAC)
O PVAC permite que qualquer terceiro verifique a correção de uma computação FHE sem acesso às chaves:
O computador (validador Octra) gera uma prova ZK (Groth16/PlonK) da avaliação correta do circuito hipergrafo
O verificador (cliente ARKHE) verifica a prova contra o commitment do circuito e o resultado cifrado
Não há necessidade de threshold decryption — privacidade preservada end-to-end
6.3.4 Limitações Identificadas
Conforme documentado no README:
"Omissions for large numbers": O código omite otimizações para números > 2^64, limitando precisão aritmética
"Excluded the transfer mechanism": Mecanismo de transferência de ciphertexts entre nós não implementado — requer camada P2P do Octra
PoC status: Proof-of-concept, não pronto para produção. Requer integração com motor de consenso ABFT

// === fhe_engine.hpp ===
#pragma once
#include <vector>
#include <cstdint>
#include <memory>

namespace pvac_hfhe {

using FpElement = __int128;
constexpr FpElement FP_PRIME = (static_cast<FpElement>(1) << 127) - 1;

class HypergraphFHE {
public:
    struct Vertex {
        uint64_t id;
        std::vector<uint8_t> ciphertext;
        double noise_level;
        uint64_t epoch;
    };

    struct Hyperedge {
        uint64_t id;
        enum GateType { AND, OR, XOR, NOT, NAND, NOR, XNOR, ADD, MUL } type;
        std::vector<uint64_t> inputs;
        uint64_t output;
        bool active;
    };

    struct Circuit {
        std::vector<Vertex> vertices;
        std::vector<Hyperedge> edges;
        double max_noise;
        double bootstrap_threshold;
    };

    std::vector<Vertex> encrypt(uint64_t value, const std::vector<uint8_t>& public_key);
    Hyperedge create_gate(GateType type, const std::vector<uint64_t>& inputs, uint64_t output);
    bool evaluate(Circuit& circuit);
    bool bootstrap(Circuit& circuit);
    std::vector<uint8_t> generate_proof(const Circuit& circuit);
    bool verify_proof(const std::vector<uint8_t>& proof, const Circuit& circuit);

private:
    std::vector<uint8_t> encrypt_bit(uint8_t bit, const std::vector<uint8_t>& pk, size_t pos);
    FpElement mod_reduce(__int128 value);
    bool check_noise(const Vertex& v);
};

// === field_ops.cpp ===
FpElement HypergraphFHE::mod_reduce(__int128 value) {
    if (value < 0) value += FP_PRIME;
    if (value < FP_PRIME) return static_cast<FpElement>(value);
    __int128 high = value >> 127;
    __int128 low = value & FP_PRIME;
    __int128 result = high + low;
    if (result >= FP_PRIME) result -= FP_PRIME;
    return static_cast<FpElement>(result);
}

// === hypergraph.cpp ===
bool HypergraphFHE::evaluate(Circuit& circuit) {
    for (auto& edge : circuit.edges) {
        switch (edge.type) {
            case Hyperedge::AND:
                edge.active = std::all_of(edge.inputs.begin(), edge.inputs.end(),
                    [&](uint64_t id) { return check_noise(circuit.vertices[id]); });
                break;
            case Hyperedge::OR:
                edge.active = std::any_of(edge.inputs.begin(), edge.inputs.end(),
                    [&](uint64_t id) { return check_noise(circuit.vertices[id]); });
                break;
            case Hyperedge::XOR: {
                int active_count = std::count_if(edge.inputs.begin(), edge.inputs.end(),
                    [&](uint64_t id) { return check_noise(circuit.vertices[id]); });
                edge.active = (active_count == 1);
                break;
            }
            case Hyperedge::NOT:
                if (!edge.inputs.empty()) {
                    edge.active = !check_noise(circuit.vertices[edge.inputs[0]]);
                }
                break;
            case Hyperedge::ADD:
            case Hyperedge::MUL:
                edge.active = evaluate_arithmetic(circuit, edge);
                break;
        }
    }
    circuit.max_noise = 0.0;
    for (const auto& v : circuit.vertices) {
        circuit.max_noise = std::max(circuit.max_noise, v.noise_level);
    }
    if (circuit.max_noise > circuit.bootstrap_threshold) {
        return bootstrap(circuit);
    }
    return true;
}

bool HypergraphFHE::bootstrap(Circuit& circuit) {
    for (auto& vertex : circuit.vertices) {
        vertex.noise_level = 0.0;
        vertex.epoch++;
    }
    circuit.max_noise = 0.0;
    return true;
}
